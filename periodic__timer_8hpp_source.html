97 template <typename F>
98 void duration_handler_impl(const time_point& last_tp, const duration& dur,
99 const std::error_code& err, F&& func) {
100 time_point now_time { Clock::now() };
101 // pass err and elapsed time to app function obj
102 if (!func(err, now_time - last_tp) ||
103 err == asio::error::operation_aborted) {
104 return; // app is finished with timer for now or timer was cancelled
105 }
106 m_timer.expires_after(dur);
107 m_timer.async_wait( [now_time, dur, f = std::move(func), this]
108 (const std::error_code& e) {
109 duration_handler_impl(now_time, dur, e, std::move(f));
110 }
111 );

116 // pass err and elapsed time to app function obj

118 err == asio::error::operation_aborted) {
